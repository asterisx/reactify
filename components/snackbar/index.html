Show Dismiss and onClose event
  • showDismiss="true"
  • onClose={() => alert("Snackbar is closed")}
Show Example

<Snackbar
    showDismiss
    message="Click on dismiss to dismiss now!"
    onClose={() => { this.setState({ showDismissSnackbar: false }); alert('Snackbar is closed'); }}
    duration={5000}
/>
