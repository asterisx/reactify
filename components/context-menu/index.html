  <div>
      Click here for menu

  </div>

    <SubMenuItem 
      onMouseDown={() => console.log('mouse down on option 1')} 
      text="Option 1" 
    />
    <SubMenuItem 
      onMouseDown={() => console.log('mouse down on option 2')} 
      text="Option 2" 
    />
  </SubMenu>
</ContextMenu>
