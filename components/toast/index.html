Show Toast
Remove Toast

<ToastProvider>
    <ToastConsumer>
    {({ addToast, removeToast }) => {
        let toastId;
        return (
        <>
            <Button
            onClick={
        () => {
            toastId = addToast(
            <Dismiss>
            Lorem Ipsum is not gypsum. Did you know Toast
            </Dismiss>,
            );
            this.setState({ toastId });
        }
        }
            >
        Show Toast
            </Button>
            <Button
            danger
            onClick={
            () => removeToast(this.state.toastId)
        }
            >
        Remove Toast
            </Button>
        </>
        );
    }
    }
    </ToastConsumer>
</ToastProvider>

      {({ addToast, removeToast }) => (
        <Button
          onClick={
            () => addToast(
              <Dismiss
                onClose={removeToast}
              >

              </Dismiss>,
            )
            }
        >
        Show Toast
        </Button>
      )}
